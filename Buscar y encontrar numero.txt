# include <iostream>
using namespace std;
int main ()
{
    int A[]={121,11,115,100};
    int c,i;
    int pos;
    cout << "que numero quiere buscar "<<endl;

    for (i=0;i<4;i++)

        if(A[i]==c)

    }

}
