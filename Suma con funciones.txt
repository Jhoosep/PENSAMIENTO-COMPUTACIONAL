#include <iostream>
using namespace std;
int suma ( int num1,int num2)
{
    return num1+num2;
}
int resta ( int num1,int num2)
{
    return num1-num2;
}
int multi ( int num1,int num2)
{
    return num1*num2;
}
int main ()
{
    int n1,n2,s,r,m;
    cin>> n1>>n2;
    s= suma (n1,n2);
    r= resta (n1,n2);
    m= multi (n1,n2);
    cout << "la suma es "<<s<< endl;
    cout << "la resta es "<<r<< endl;
    cout << "la multiplicacion es "<<m<< endl;
    return 0;
}
